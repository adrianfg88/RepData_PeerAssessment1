fill_in <- function(x,y)
     {
     #browser()
          for (i in 1:nrow(x)) 
          {
               if (is.na(x[i,1]))
               {
                   x[i,1]<-y[y[,1] == (x[i,3]),2] 
               }
          }
     x
     }
